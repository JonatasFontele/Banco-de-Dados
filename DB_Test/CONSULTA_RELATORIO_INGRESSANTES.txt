
SELECT "LY_ALUNO"."ALUNO", "LY_ALUNO"."NOME_COMPL", "LY_MATRICULA"."SIT_MATRICULA", "LY_MATRICULA"."ANO", "LY_MATRICULA"."SEMESTRE", "LY_ALUNO"."CURSO", "LY_CURSO"."NOME", "LY_TURNO"."TURNO", "LY_CURSO"."CURSO", "LY_ALUNO"."TURNO", "LY_ALUNO"."ANO_INGRESSO", "LY_ALUNO"."SEM_INGRESSO"
 FROM   ((("LY_MATRICULA" "LY_MATRICULA" INNER JOIN "LY_ALUNO" "LY_ALUNO" ON "LY_MATRICULA"."ALUNO"="LY_ALUNO"."ALUNO") INNER JOIN "LY_CURSO" "LY_CURSO" ON "LY_ALUNO"."CURSO"="LY_CURSO"."CURSO") INNER JOIN "Lyceum"."dbo"."LY_CURRICULO" "LY_CURRICULO" ON (("LY_ALUNO"."CURSO"="LY_CURRICULO"."CURSO") AND ("LY_ALUNO"."TURNO"="LY_CURRICULO"."TURNO")) AND ("LY_ALUNO"."CURRICULO"="LY_CURRICULO"."CURRICULO")) INNER JOIN "Lyceum"."dbo"."LY_TURNO" "LY_TURNO" ON "LY_CURRICULO"."TURNO"="LY_TURNO"."TURNO"
 WHERE  "LY_ALUNO"."ANO_INGRESSO"=2019 AND "LY_ALUNO"."SEM_INGRESSO"=2 AND "LY_MATRICULA"."SIT_MATRICULA"='Matriculado' AND "LY_MATRICULA"."ANO"=2021 AND "LY_MATRICULA"."SEMESTRE"=1 AND "LY_ALUNO"."CURSO"='11'
 ORDER BY "LY_CURSO"."CURSO", "LY_ALUNO"."TURNO", "LY_ALUNO"."NOME_COMPL"
